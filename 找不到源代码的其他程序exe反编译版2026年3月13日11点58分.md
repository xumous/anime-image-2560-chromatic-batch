def parse_m3u_content(content):
    """
    解析 M3U 格式内容，提取频道信息
    返回: [{"id": "", "name": "", "url": "", "logo": "", "group": ""}, ...]
    """
    channels = []
    lines = content.strip().split('\n')
    current_channel = None
    for line in lines:
        line = line.strip()
        if not line:
            continue
        if line.startswith('#EXTINF'):
            current_channel = {'id': '', 'name': '', 'url': '', 'logo': '', 'group': ''}
            id_match = re.search('tvg-id="([^"]*)"', line)
            if id_match:
                current_channel['id'] = id_match.group(1)
            name_match = re.search('tvg-name="([^"]*)"', line)
            if name_match:
                current_channel['name'] = name_match.group(1)
            logo_match = re.search('tvg-logo="([^"]*)"', line)
            if logo_match:
                current_channel['logo'] = logo_match.group(1)
            group_match = re.search('group-title="([^"]*)"', line)
            if group_match:
                current_channel['group'] = group_match.group(1)
            comma_pos = line.rfind(',')
            if comma_pos > 0:
                display_name = line[comma_pos + 1:].strip()
                if display_name and not current_channel['name']:
                    current_channel['name'] = display_name
        else:
            if line.startswith('http') and current_channel is not None:
                current_channel['url'] = line
                channels.append(current_channel)
                current_channel = None
    return channels

                    if self.ffmpeg_process.stderr:
                        self.ffmpeg_process.stderr.close()

            self.stderr_thread = threading.Thread(target=read_stderr, daemon=True)
            self.stderr_thread.start()

            time.sleep(5)
            if self.ffmpeg_process.poll() is not None:
                self.stderr_thread.join(timeout=1)
                self.finished_signal.emit(False, f'ffmpeg 启动后立即退出，返回码 {self.ffmpeg_process.returncode}')
                return

            start_time = time.time()
            last_check = time.time()
            last_size = 0
            stall_count = 0
            stall_threshold = 6

            while self.ffmpeg_process.poll() is None and not self.is_stopped:
                elapsed = time.time() - start_time
                if elapsed >= self.duration:
                    break
                progress = 10 + int(elapsed / self.duration * 90)
                self.progress_signal.emit(min(progress, 99))

                if time.time() - last_check > 5:
                    if os.path.exists(self.output_file):
                        current_size = os.path.getsize(self.output_file)
                        size_mb = current_size / 1048576
                        self.log_signal.emit(f'[INFO] 当前文件大小: {size_mb:.2f} MB')
                        if current_size == last_size:
                            stall_count += 1
                            if stall_count >= stall_threshold:
                                self.log_signal.emit('[ERROR] 录制停滞：文件大小长时间未增长，自动停止')
                                self.ffmpeg_process.terminate()
                                try:
                                    self.ffmpeg_process.wait(timeout=5)
                                except:
                                    self.ffmpeg_process.kill()
                                self.finished_signal.emit(False, '录制停滞：文件大小长时间未增长')
                                return
                        else:
                            stall_count = 0
                            last_size = current_size
                    else:
                        self.log_signal.emit('[WARN] 输出文件尚未创建')
                    last_check = time.time()

                time.sleep(0.5)

            if self.is_stopped:
                self.ffmpeg_process.terminate()
                try:
                    self.ffmpeg_process.wait(timeout=5)
                except:
                    self.ffmpeg_process.kill()
                self.finished_signal.emit(False, '录制已停止')
            else:
                stdout, stderr = self.ffmpeg_process.communicate(timeout=30)
                if self.ffmpeg_process.returncode == 0:
                    self.progress_signal.emit(100)
                    self.finished_signal.emit(True, f'录制完成: {self.output_file}')
                else:
                    error_msg = stderr[-500:] if stderr else '未知错误'
                    self.finished_signal.emit(False, f'录制失败: {error_msg}')
        except Exception as e:
            self.finished_signal.emit(False, f'发生错误: {str(e)}')


class NetworkTestThread(QThread):
    """网络测试线程"""
    result_signal = pyqtSignal(bool, str)

    def __init__(self, url):
        super().__init__()
        self.url = url

    def run(self):
        """测试网络连接，使用 GET 请求并读取前 1024 字节"""
        import urllib.request
        import socket
        socket.setdefaulttimeout(15)

        try:
            req = urllib.request.Request(self.url, method='GET')
            req.add_header('User-Agent', 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36')
            req.add_header('Range', 'bytes=0-1023')
            response = urllib.request.urlopen(req, timeout=15)
            data = response.read(1024)
            status = response.getcode()
            if status == 200 or status == 206:
                self.result_signal.emit(True, f'连接成功 (HTTP {status})，已读取 {len(data)} 字节')
            else:
                self.result_signal.emit(True, f'连接正常 (HTTP {status})')
        except urllib.error.HTTPError as e:
            if e.code in [401, 403, 405]:
                self.result_signal.emit(True, f'服务器有响应 (HTTP {e.code})')
            else:
                self.result_signal.emit(False, f'HTTP 错误 {e.code}')
        except Exception as e:
            # 尝试不带Range头的简单请求
            try:
                req = urllib.request.Request(self.url, method='GET')
                req.add_header('User-Agent', 'Mozilla/5.0')
                response = urllib.request.urlopen(req, timeout=15)
                data = response.read(1024)
                self.result_signal.emit(True, f'连接成功 (HTTP {response.getcode()})，已读取 {len(data)} 字节')
            except Exception as e2:
                self.result_signal.emit(False, f'连接失败: {str(e2)}')

    def generate_auto_filename(self):
        """生成自动文件名"""
        channel_data = self.channel_combo.currentData()
        if channel_data:
            channel_id = channel_data['id']
        else:
            channel_id = 'Unknown'
        start_dt = self.start_time_edit.dateTime().toPyDateTime()
        end_dt = self.end_time_edit.dateTime().toPyDateTime()
        start_str = start_dt.strftime('%Y-%m-%d_%H-%M-%S')
        end_str = end_dt.strftime('%Y-%m-%d_%H-%M-%S')
        filename = f'{channel_id}_{start_str}—{end_str}_录制.ts'
        save_dir = self.get_app_dir()
        output_path = self.get_unique_filename(os.path.join(save_dir, filename))
        self.output_edit.setText(output_path)

    def get_unique_filename(self, filepath):
        """获取唯一的文件名（处理重名）"""
        if not os.path.exists(filepath):
            return filepath
        directory = os.path.dirname(filepath)
        filename = os.path.basename(filepath)
        name, ext = os.path.splitext(filename)
        counter = 1
        while True:
            new_filename = f'{name}_{counter}{ext}'
            new_filepath = os.path.join(directory, new_filename)
            if not os.path.exists(new_filepath):
                return new_filepath
            counter += 1

            QMessageBox.warning(self, '输入错误', '结束时间必须晚于开始时间！')

            QMessageBox.warning(self, '输入错误', f'无法创建输出目录: {e}')

def crop_center_square(img):
    """从图片中心裁剪正方形"""
    width, height = img.size
    size = min(width, height)
    left = (width - size) // 2
    top = (height - size) // 2
    right = left + size
    bottom = top + size
    return img.crop((left, top, right, bottom))


def create_collage(folder_path):
